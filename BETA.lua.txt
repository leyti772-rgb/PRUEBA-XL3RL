-- 1. FUNCI√ìN OBLIGATORIO (No-Clip)
local noClipEnabled = false
local noClipConnection = nil
local ObligatorioBoton = crearBoton("OBLIGATORIO: OFF", function(boton)
    noClipEnabled = not noClipEnabled
    ObligatorioBoton.Text = noClipEnabled and "OBLIGATORIO: ON" or "OBLIGATORIO: OFF"
    ObligatorioBoton.BackgroundColor3 = noClipEnabled and Config.ColorPrincipal or Color3.fromRGB(50, 50, 50)

    if noClipEnabled then
        noClipConnection = RunService.Stepped:Connect(function()
            local char = player.Character
            if char then for _, part in ipairs(char:GetDescendants()) do if part:IsA("BasePart") and part.CanCollide then part.CanCollide = false end end end
        end)
    else
        if noClipConnection then noClipConnection:Disconnect(); noClipConnection = nil end
    end
end)

-- 2. FUNCI√ìN PARA EJECUTAR SCRIPTS EXTERNOS
local function executeScript(button, url)
    spawn(function()
        local originalText = button.Text
        button.Text = "Cargando..."
        button.BackgroundColor3 = Color3.fromRGB(200, 150, 0)
        
        local success, result = pcall(function() loadstring(game:HttpGet(url))() end)
        
        if success then
            button.Text = "¬°Ejecutado!"
            button.BackgroundColor3 = Config.ColorPrincipal
            wait(2)
            button.Text = originalText
            button.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
        else
            button.Text = "¬°Error!"
            button.BackgroundColor3 = Color3.fromRGB(200, 0, 0)
            warn("Error ejecutando script: " .. tostring(result))
            wait(2)
            button.Text = originalText
            button.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
        end
    end)
end
