-- FUNCIÓN PARA CREAR BOTONES
local function crearBoton(texto, callback)
    local Boton = Instance.new("TextButton")
    Boton.Size = UDim2.new(1, -20, 0, 35); Boton.Text = texto; Boton.Font = Enum.Font.SourceSans; Boton.TextSize = 16; Boton.TextColor3 = Color3.new(1, 1, 1); Boton.BackgroundColor3 = Color3.fromRGB(50, 50, 50); Boton.Parent = ButtonContainer
    local BotonCorner = Instance.new("UICorner"); BotonCorner.Parent = Boton
    local BotonStroke = Instance.new("UIStroke"); BotonStroke.Color = Color3.fromRGB(70, 70, 70); BotonStroke.Parent = Boton
    Boton.MouseButton1Click:Connect(callback)
    return Boton
end

-- LÓGICA DE DRAG Y BOTONES
local dragging = false; local dragStart; local startPos
Header.InputBegan:Connect(function(input) if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then dragging = true; dragStart = input.Position; startPos = MainContainer.Position end end)
Header.InputEnded:Connect(function(input) if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then dragging = false end end)
UserInputService.InputChanged:Connect(function(input) if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then local delta = input.Position - dragStart; MainContainer.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y) end end)
CloseButton.MouseButton1Click:Connect(function() ScreenGui:Destroy() end)
MinButton.MouseButton1Click:Connect(function() MainContainer.Visible = not MainContainer.Visible end)

-- ================== LÓGICA DE FUNCIONES ==================
local noClipEnabled = false; local noClipConnection = nil
local ObligatorioBoton = crearBoton("OBLIGATORIO: OFF", function(boton)
    noClipEnabled = not noClipEnabled
    ObligatorioBoton.Text = noClipEnabled and "OBLIGATORIO: ON" or "OBLIGATORIO: OFF"
    ObligatorioBoton.BackgroundColor3 = noClipEnabled and Config.ColorPrincipal or Color3.fromRGB(50, 50, 50)
    if noClipEnabled then noClipConnection = RunService.Stepped:Connect(function() local char = player.Character; if char then for _, part in ipairs(char:GetDescendants()) do if part:IsA("BasePart") and part.CanCollide then part.CanCollide = false end end end end) else if noClipConnection then noClipConnection:Disconnect(); noClipConnection = nil end end
end)

local function executeScript(button, url)
    spawn(function()
        local originalText = button.Text
        button.Text = "Cargando..."; button.BackgroundColor3 = Color3.fromRGB(200, 150, 0)
        local success, result = pcall(function() loadstring(game:HttpGet(url))() end)
        if success then
            button.Text = "¡Ejecutado!"; button.BackgroundColor3 = Config.ColorPrincipal
            wait(2); button.Text = originalText; button.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
        else
            button.Text = "¡Error!"; button.BackgroundColor3 = Color3.fromRGB(200, 0, 0)
            warn("Error ejecutando script: " .. tostring(result))
            wait(2); button.Text = originalText; button.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
        end
    end)
end
